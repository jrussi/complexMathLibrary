complexMathLib.js

//define two random complex numbers
var cmpx1 = new Complex(Math.random().toFixed(1)*10, Math.random().toFixed(1)*10);
var cmpx2 = new Complex(Math.random().toFixed(1)*10, Math.random().toFixed(1)*10);
//the result will be the sum of the first two vars
var cmpx3 = cmpx1.calcSum(cmpx2);
//append results to element ex1
ex1.appendChild(formatMatrix([[cmpx1]]));
ex1.appendChild(document.createTextNode("+"));
ex1.appendChild(formatMatrix([[cmpx2]]));
ex1.appendChild(document.createTextNode("="));
ex1.appendChild(formatMatrix([[cmpx3]]));



complexMatrixLib.js

//define 4 numbers
var cmpx11 = new Complex(4,5);
var cmpx12 = new Complex(2,6);
var cmpx21 = new Complex(4,4);
var cmpx22 = new Complex(1,5);
//create matrix
var mat = [[cmpx11, cmpx12], [cmpx21, cmpx22]];
//calculate the determinant
var matDet = mat.calcDeterminant();
//append results to element ex2
ex2.appendChild(document.createTextNode("Determinant("));
ex2.appendChild(formatMatrix(mat);
ex2.appendChild(document.createTextNode(") = "));
ex2.appendChild(formatMatrix(matDet));
